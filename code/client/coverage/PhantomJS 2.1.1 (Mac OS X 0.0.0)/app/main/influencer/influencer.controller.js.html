(function() {
  'use strict';
 
  angular
    .module('ppulse')
    .controller('InfluencerController', InfluencerController);
 
  /** @ngInject */
  function InfluencerController($scope, CommanService, authService, $timeout, $log, $mdDialog, $sce, toastr, $window, InfluencerService) {
    var vm = this,
        user = angular.copy(authService.getUser());
    vm.commanService = CommanService;
    vm.influencerService = InfluencerService;
    vm.getInfluencers = getInfluencers;
 
    vm.init = init;
 
    if(!user){ $state.go('login'); } else { vm.init(); }
 
    vm.processing = true;
    vm.timePeriod = 'Last 7 Days';
    vm.playAnim = true;
    vm.influencers = [];
 
    function getAlerts() {
      vm.commanService.getAlerts(user.token).then(function(res){
        vm.alerts = angular.copy(res.data.ppulse.data);
        vm.alertid = vm.alerts[0].alertid;
        getInfluencers();
      }).catch(function(err){
        $log.log(err);
      });
    }
 
    function getCities() {
      vm.commanService.getCities().then(function(res){
        vm.cities = angular.copy(res.data.ppulse.data);
      }).catch(function(err){
        $log.log(err);
      });
    }
 
    function getInfluencers() {
      vm.processing = true;
      var todayDate = new Date();
 
      var query = {
        alertid : vm.alertid,
        idate : undefined,
        edate : todayDate
      };
 
      if(vm.timePeriod === 'Last Day') {
        query.idate = moment().subtract(1, 'days')._d;
      } else if(vm.timePeriod === 'Last 7 Days') {
        query.idate = moment().subtract(7, 'days')._d;
      } else if(vm.timePeriod === 'Last Month') {
        query.idate = moment().subtract(30, 'days')._d;
      }
 
      vm.influencerService.getTwitterInfluencersPage(query, user.token).then(function(res){
        console.log(res.data.ppulse.data.influencers);
        vm.influencers = res.data.ppulse.data.influencers;
        vm.processing = false;
      }).catch(function(err){
        $log.log(err);
      });
    }
 
    function init() {
      getCities();
      getAlerts();
 
    }
 
  }
})();
